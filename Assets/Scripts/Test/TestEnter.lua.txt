
function start()
    print('testenter-start')
    local tab = _ENV["testentertable"]
    if(tab == nil) then print("Can't find testentertable") end
    
    for k,v in pairs(tab) do
        local st,ed = string.find(k,"button")
        if(st ~= nil and ed == string.len(k))  then
            local funcName = "on"..k.."click"
            local func = _ENV[funcName]
            if(func ~= nil) then
                tab[k].onClick:AddListener(_ENV[funcName])
                print("bind succeed")
            else
                print("Can Find Component but Func is nil")
                print(k .."    ".. funcName)
            end
            goto continue
        end
        st,ed = string.find(k,"slider")
        
        if(st ~= nil and ed == string.len(k))  then
            local funcName = "on"..k.."valuechanged"
            local func = _ENV[funcName]
            if(func ~= nil) then
                tab[k].onValueChanged:AddListener(_ENV[funcName])
                print("bind succeed")
            else
                print("Can Find Component but Func is nil")
                print(k .."    ".. funcName)
            end
            goto continue
        end
        ::continue::
    end
    
end

function onaddbuttonclick()
    print("addclicked")
    local tab = _ENV["testentertable"]
    local TMP = tab['shownumbertmp']:GetComponent("TextMeshProUGUI")
    if(TMP == nil) then
        print("NumberNil")
        print(type(TMP))
    end
    TMP.text = TMP.text + 1
end
function onsubbuttonclick()
    print("subclicked")

    local TMP = tab['shownumbertmp']:GetComponent("TextMeshProUGUI")
    if(TMP == nil) then
        print("NumberNil")
    end
    TMP.text = TMP.text - 1
   
end
